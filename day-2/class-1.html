git config --global user.name 'Name' - add the user name

git config --global user.email 'ouremail' - Add the Email

git init - initialize the repository/project

git add . - add all the changes to the repository

git add file-name - add the changes that are made in that file to the repository

git commit -m 'Added a button/paragraph/Page'

git remote add origin folder/file.git - Add remote repository as origin

git branch -M master/main - create a main/master branch

git branch homepage - create a branch with name homepage

git merge aboutpage - Merge the aboutpage branch with current branch

git checkout aboutpage - Switch to a specific branch

git branch - show all the branches in our origin/repository

git push -u origin master - pushes the code in the master branch

git push -u origin branch_name - push the code to the repository

git pull - pull/update the code in our device after merging

git fetch - pull/update the code without merging

git clone url - clone the project

git log - show the commit history

git log --oneline - Show commit history in a condensed manner

git remote -shows origin is added or not

git remote -v - show the remote repository
